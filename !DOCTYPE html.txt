  <title>แผนที่จาก Google Sheets</title>
  <meta charset="utf-8" />

  <link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css" />
  <style>
    #map { height: 100vh; }
  </style>
</head>
<body>

<div id="map"></div>

<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
<script>
  const map = L.map('map').setView([13.7563, 100.5018], 13); // ตั้งค่าพิกัดเริ่มต้นและซูม

  L.tileLayer('https://tile.openstreetmap.org/{z}/{x}/{y}.png', {
    attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
  }).addTo(map);

  // URL ของ Google Apps Script Web App ที่ดึงข้อมูลจาก Google Sheets
  const googleSheetUrl = 'https://script.google.com/macros/s/AKfycbxoOtTYuTtahxsc-7YVa2zeY82SZRUu2gsWwED4OOXFDOnVZZLyUlvll0wPRRuFQiO9/exec';

  fetch(googleSheetUrl)
    .then(response => response.json())
    .then(data => {
      data.forEach(item => {
        // ตรวจสอบชื่อคอลัมน์ให้ตรงกับ Google Sheets
        const lat = parseFloat(item['ละติจูต']);
        const lng = parseFloat(item['ลองจิจูด']);

        // ตรวจสอบว่าข้อมูลพิกัดถูกต้อง
        if (!isNaN(lat) && !isNaN(lng)) {
          // สร้างข้อความ Pop-up โดยใช้ข้อมูลจากคอลัมน์อื่น ๆ
          const popupContent = `
            <b>${item['ชื่อร้านค้า']}</b><br>
            ประเภท: ${item['ประเภทตลาด']}<br>
            สินค้า: ${item['สินค้าที่ขาย']}<br>
            ราคา: ${item['ราคา']} บาท
          `;
          
          L.marker([lat, lng]).addTo(map)
            .bindPopup(popupContent);
        }
      });
    })
    .catch(error => {
      console.error('Error fetching data:', error);
    });
</script>
</body>
</html>
